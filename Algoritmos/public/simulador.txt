    function analisar() {
      const ctx = document.getElementById('myChart1');
      var logradouro = inp_logradouro.value
      var bairro = inp_bairro.value

      if (logradouro == "" && bairro == "") {
        alert("Informe o logradouro e o bairro")
      } else if (logradouro == "") {
        alert("Informe o logradouro")
      } else if (bairro == "") {
        alert("Informe o bairro")
      } else {
            new Chart(ctx, {
          type: 'bar',
          data: {
            labels: ['18 a 25', '25 a 40', '40 a 55', '55 a 70', 'mais de 70'],
            datasets: [{
              label: 'Mulheres',
              data: [9, 9, 7, 4, 2, 0],
              borderWidth: 1,
              backgroundColor: "#fe6383"
            },
            {
              label: 'Homens',
              data: [10, 8, 5, 2, 3, 1],
              borderWidth: 1,
              backgroundColor: "#37a2ea"
            }
          ]
          },
          options: {
             plugins: {
                  title: {
                      display: true,
                      text: `Ocupação diaria: ${logradouro}`,
                      font: {size: 18}
                  }
              },
            scales: {
              y: {
                beginAtZero: true
              }
            }
          }
        });
      }
    }
